# Build and publish

ImageViewerX/Src/ImageViewer.Desktop$ dotnet publish ImageViewer.Desktop.csproj -c Release -r linux-x64 -f net9.0 -p:PublishAot=true

# Grab published binaries

ImageViewerX/Src/ImageViewer.Desktop/bin/Release/net9.0/linux-x64/publish/libHarfBuzzSharp.so
ImageViewerX/Src/ImageViewer.Desktop/bin/Release/net9.0/linux-x64/publish/libSkiaSharp.so
ImageViewerX/Src/ImageViewer.Desktop/bin/Release/net9.0/linux-x64/publish/ImageViewerX
ImageViewerX/Src/ImageViewer.Desktop/bin/Release/net9.0/linux-x64/publish/Assets

# Put them in ...

ImageViewerX/Src/deb-staging/usr/lib/ImageViewerX/


(Make sure "ImageViewerX/Src/deb-staging/usr/bin/ImageViewerX" shell script has "Executable as Program" permission)

# Create a deb package

ImageViewerX/Src/ImageViewerX.Desktop$ dpkg-deb --root-owner-group --build ./deb-staging ./ImageViewerX_1.0.0_amd64.deb

# Test it

ImageViewerX/Src/ImageViewerX.Desktop$ sudo apt install ./ImageViewerX_1.0.0_amd64.deb
